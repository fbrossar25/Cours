	\begin{itemize}
		\item cas d'arrêt :	
			$$i \geqslant n : \underset{\to 0}{\text{plus d'élément}}$$
		\item si $i < n$ et $T(i+1) < m$
		\begin{align*}
		&\implies T(i+1) \notin SSA\\
		&\implies \text{on continue avec } T(i+2) \implies LSSA(i,m) = LSSA(i+1, m)
		\end{align*}
		\item si $i < n$ et $T(i + 1) \geqslant n$ :
			$$\text{max}\begin{cases}
			T(i+1) \in SSA \implies LSSA(i,m) = LSSA(i+1, T(i+1)) + 1\\
			T(i+1) \notin SSA \implies LSSA(i,m) = LSSA(i+1, m)
			\end{cases}$$
	\end{itemize}

	\subsection{Question 4}
	\begin{algorithm}
		\begin{algorithmic}[1]%numéroter toute les n lignes
			\State $\text{tableau L initialisé à -1}$
			\Function{LSSA\_memo}{$i,j$}
			\If {$L(i,j) == -1$}
				\If {$i = n$}
					\State $L(i,j) = 0$
				\ElsIf {$T(i+1) < T(j)$}
					\State $L(i,j) = LSSA\_memo(i+1,j)$
				\Else
					\State $L(i,j) = Max[LSSA\_memo(i+1,j), 1+LSSA\_memo(i+1,i+1)]$
				\EndIf
			\EndIf
			\State \Return $L(i,j)$
			\EndFunction
		\end{algorithmic}
	\end{algorithm}
