public class tableau{
	final int CAPACITE;
	int nbElements;
	int[] tab;
	tableau(int n){
		CAPACITE = n;
		tab = new int[CAPACITE];
	}
	tableau(){
		tableau(100);
	}
	boolean estVide(){
		return (taille == 0);
	}
	boolean estPlein(){
		return (taille == CAPACITE);
	}
	boolean estPresent(int n){
		boolean presence = false;
		int i = 0;
		while(!presence && (i < taille)){
			if(t[i] == n)presence <- true;
			else i++;
		}
		return presence
	}
	ajout(int n){
		if(!estPlein()){
			t[taille] = n;
			taille++;
		}
		else Ecran.afficher("tableau plein");
	}
	retirer(int n){
		int i = 0;
		boolean stop = false;
		while((i < taille) && !stop){
			if(t[i]==n){
				stop = true;
				int j;
				for(j=i;j<taille;j++){
					t[j] = t[j+1];
				}
			}
			else i++;
		}
	}
	retirerTout(int n){
		int i=0,j;
		while(i < taille){
			if(t[i] == n){
				for(j=i;j<taille;j++){
					t[j] = t[j+1];
				}
			}
			else i++;
		}
	}
}
